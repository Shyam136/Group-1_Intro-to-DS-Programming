# All of the instructions below are to be run at the terminal within your VSCode environment

# get the latest code
git pull origin main


# build the image
docker build -t group1app .


# run the image
docker rm -f group1app_container
docker run -d -p 8501:8501 --name group1app_container group1app


# view the application in the browser
http://0.0.0.0:8501/

# stop the docker container
docker stop group1app_container
